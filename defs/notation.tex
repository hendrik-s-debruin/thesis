% ==============================================================================
% Relation Operator
% ==============================================================================
	\newcommand{\relop}{\ensuremath{\bowtie}}
	\newglossaryentry{not:relop}
	{%
		name=\ensuremath{\relop},
		description=an operator in the set \ensuremath{\{<, \leq, =, \geq, >\}},
		type=notation
	}
	\glsadd{not:relop}
